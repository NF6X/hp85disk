234 void gpib_bus_read_init(int busy)
235 {
236  // Switch GPIB control lines and data bus to an inactive state
237  // Do it in such a way as to avoid CPU and GPIB transceiver direction conflicts
238  // Avoid setting any GPIB BUS/Control LOW unintensionally, while switching, even for a moment
239  // Notes - when all GPIB bus control lines are HIGH they are basically inactive
240  // Release IFC and REN
241  GPIB_PIN_FLOAT_UP(IFC); // IFC FLOAT PULLUP
242  GPIB_PIN_FLOAT_UP(REN); // REN FLOAT PULLUP
243 
244 #if BOARD == 2
245  // We are NOT a controller
246  GPIB_IO_LOW(SC); // REN IN, IFC IN, LOW ALWAYS for a Device
247 #endif
248 
249  // Avoid transceiver direction conflict so set input and pullups
250  GPIB_BUS_IN(); // CPU data direction IN
251  GPIB_BUS_LATCH_WR(0xff);// Float BUS pins HIGH
252 
253 #if BOARD == 2
254  // OC GPIB data BUS driver mode
255  // Notes:
256  // If we were WRITTING - 0xff and OC drivers float the bus
257  // If we will be READING this is harmless
258  GPIB_IO_LOW(PE); // Enable OC GPIB BUS transceivers
259 #endif
260 
261  GPIB_PIN_FLOAT_UP(DAV); // DAV FLOAT PULLUP
262  GPIB_PIN_FLOAT_UP(ATN); // ATN FLOAT PULLUP
263  GPIB_PIN_FLOAT_UP(EOI); // EOI FLOAT PULLUP
264  GPIB_PIN_FLOAT_UP(SRQ); // SRQ FLOAT PULLUP
265 
266  if(!busy)
267  {
272  GPIB_PIN_FLOAT_UP(NRFD); // OC PULLUP
273  GPIB_PIN_FLOAT_UP(NDAC); // OC PULLUP
274  GPIB_BUS_SETTLE(); // Let OC signals go HI and settle
275  }
276  else
277  {
278  GPIB_IO_LOW(NDAC);
279  GPIB_IO_LOW(NRFD);
280  }
281 
282 #if BOARD == 2
283  // Switch GBIB driver direction
284  GPIB_IO_LOW(TE); // BUS IN, DAV IN, NDAC OUT, NRFD OUT
285  GPIB_IO_HI(DC); // ATN IN, EOI IN, SRQ OUT OC
286 #endif
287 
288  GPIB_BUS_SETTLE(); // Let Data BUS settle
289 }

632 uint16_t gpib_write_byte(uint16_t ch)
633 {
634  uint8_t tx_state;
635 
636  // Wait for DAV to be released before starting
637  // Read state for DAV
638  gpib_bus_read_init(0);
639 
640  // Bus read init sets the following states
641  // GPIB_BUS_LATCH_WR(0xff); // float OC BUS on write
642  // GPIB_IO_LOW(PE); // BUS OC PULLUP
643 
644  tx_state = GPIB_TX_START;
645  gpib_timeout_set(HTIMEOUT);
646 
647  while(tx_state != GPIB_TX_DONE )
648  {
649  if(uart_keyhit(0))
650  break;
651 
652 #if 0
653  // FIXME - this is disabled as it breaks write
654  // Try to detect PPR - only for debugging
655  if(gpib_detect_PP())
656  ch |= PP_FLAG;
657 #endif
658 
659  // IFC is always in for a device
660  if(GPIB_PIN_TST(IFC) == 0)
661  {
662  ch |= IFC_FLAG;
663  gpib_bus_init();
664  break;
665  }
666 
667  switch(tx_state)
668  {
669  case GPIB_TX_START:
670  // Wait for release of DAV and EOI before starting
671  // IF DAV = 0 the bus is busy
672  if(GPIB_PIN_TST(DAV) == 1)
673  {
674 #if BOARD == 2
675  // We arive here in Read Mode - see gpib_bus_read_init(0);
676 
677  // Switch GBIB driver direction to Write
678  // CPU has BUS already set IN with pullups = 0xff - see gpib_bus_read_init(0);
679  // When we switch the GPIB drivers will be in OC mode and off the bus (0xff)
680  GPIB_IO_HI(TE); // BUS OUT, DAV OUT, NRFD and NDAC IN
681  GPIB_IO_LOW(DC);// ATN OUT, EOI OUT, SRQ IN
682 #endif
683  // My testing with various GPIB devices shows that we MUST assert ATN EARLY!
684  if(ch & ATN_FLAG)
685  GPIB_IO_LOW(ATN); // FYI: SS80 never sends ATN from a device
686  else
687  GPIB_PIN_FLOAT_UP(ATN);
688 
689  gpib_timeout_set(HTIMEOUT);
690  tx_state = GPIB_TX_WAIT_READY;
691  }
692  if (gpib_timeout_test())
693  {
694 #if SDEBUG
695  if(debuglevel & (1+4))
696  printf("<BUS waiting for DAV==1>\n");
697 #endif
698  ch |= TIMEOUT_FLAG;
699  tx_state = GPIB_TX_ERROR;
700  break;
701  }
702  break;
703 
704  case GPIB_TX_WAIT_READY:
705  // Wait for ready condition
706  if(GPIB_PIN_TST(NRFD) == 1 && GPIB_PIN_TST(NDAC) == 0)
707  {
708  gpib_timeout_set(HTIMEOUT);
709  tx_state = GPIB_TX_PUT_DATA;
710  }
711  if (gpib_timeout_test())
712  {
713 #if SDEBUG
714  if(debuglevel & (1+4))
715  printf("<BUS waiting for NRFD==1 && NDAC == 0>\n");
716 #endif
717  ch |= TIMEOUT_FLAG;
718  tx_state = GPIB_TX_ERROR;
719  break;
720  }
721  break;
722 
723  case GPIB_TX_PUT_DATA:
724 
725  if(ch & EOI_FLAG)
726  GPIB_IO_LOW(EOI);
727  else
728  GPIB_PIN_FLOAT_UP(EOI);
729 
730  GPIB_BUS_WR((ch & 0xff) ^ 0xff); // Write Data inverted
731 #if BOARD == 2
732  // Switch to Tristate mode to soeed up Write
733  GPIB_IO_HI(PE);
734 #endif
735  GPIB_BUS_SETTLE(); // Let Data BUS settle
736 
737  gpib_timeout_set(HTIMEOUT);
738  tx_state = GPIB_TX_SET_DAV_LOW;
739  break;
740 
741 
742  case GPIB_TX_SET_DAV_LOW:
743  GPIB_IO_LOW(DAV);
744  gpib_timeout_set(HTIMEOUT);
745  tx_state = GPIB_TX_WAIT_FOR_NRFD_LOW;
746  break;
747 
749  case GPIB_TX_WAIT_FOR_NRFD_LOW:
750  if (GPIB_PIN_TST(NRFD) == 0)
751  {
752  gpib_timeout_set(HTIMEOUT);
753  tx_state = GPIB_TX_WAIT_FOR_NDAC_HI;
754  break;
755  }
756  if (gpib_timeout_test())
757  {
758  ch |= TIMEOUT_FLAG;
759  tx_state = GPIB_TX_ERROR;
760 #if SDEBUG
761  if(debuglevel & (1+4))
762  printf("<BUS waiting for NRFD==0>\n");
763 #endif
764  break;
765  }
766  break;
767 
769  case GPIB_TX_WAIT_FOR_NDAC_HI:
770  if(GPIB_PIN_TST(NDAC) == 1) // Byte byte accepted
771  {
772  tx_state = GPIB_TX_SET_DAV_HI;
773  break;
774  }
775  if (gpib_timeout_test())
776  {
777  ch |= TIMEOUT_FLAG;
778  tx_state = GPIB_TX_ERROR;
779 #if SDEBUG
780  if(debuglevel & (1+4))
781  printf("<BUS waiting for NDAC==1>\n");
782 #endif
783  break;
784  }
785  break;
786 
788  case GPIB_TX_SET_DAV_HI:
789  GPIB_PIN_FLOAT_UP(DAV);
790  GPIB_BUS_SETTLE(); // give some time
791 
792  gpib_bus_read_init(0); // Free BUS, NOT busy
793 
794  gpib_timeout_set(HTIMEOUT);
795  tx_state = GPIB_TX_WAIT_FOR_DAV_HI;
796  break;
797 
800  case GPIB_TX_WAIT_FOR_DAV_HI:
801  if(GPIO_PIN_TST(DAV) == 1)
802  {
803  tx_state = GPIB_TX_FINISH;
804  break;
805  }
806  if (gpib_timeout_test())
807  {
808  ch |= TIMEOUT_FLAG;
809  tx_state = GPIB_TX_ERROR;
810 #if SDEBUG
811  if(debuglevel & (1+4))
812  printf("<BUS waiting for DAV==1>\n");
813 #endif
814 
815  }
816  break;
817 
818  case GPIB_TX_FINISH:
819  tx_state = GPIB_TX_DONE;
820  break;
821 
822  case GPIB_TX_ERROR:
823 #if SDEBUG
824  if(debuglevel & (1+4))
825  printf("<NRFD=%d,NDAV=%d>\n", GPIB_PIN_TST(NRFD),GPIB_PIN_TST(NDAC));
826 #endif
827 // Free BUS, BUSY on error
828  gpib_bus_read_init(1);
829  tx_state = GPIB_TX_DONE;
830  break;
831 
832  // FIXME do we want to be busy at this point
833  case GPIB_TX_DONE:
834  break;
835  }
836  }
837  return(ch);
838 }
