697 uint16_t gpib_read_byte( void )
698 {
699  uint16_t val,status,ch;
700  uint8_t rx_state;
701 
702  ch = 0;
703  status = 0;
704 
705  GPIB_PIN_FLOAT(IFC);
706  GPIB_PIN_FLOAT(ATN);
707  GPIB_PIN_FLOAT(EOI);
708  GPIB_PIN_FLOAT(DAV); // DAV should be HI already
709 
710 
711  GPIB_BUS_IN();
712 
713  if(gpib_unread_f)
714  {
715  gpib_unread_f = 0;
716  return(gpib_unread_data);
717  }
718 
719  GPIB_PIN_FLOAT(NRFD);
720  GPIB_BUS_SETTLE(); // Let Data BUS settle
721  GPIB_IO_LOW(NDAC);
722 
723  rx_state = GPIB_RX_START;
724 
725  while(rx_state != GPIB_RX_DONE)
726  {
727  if(Parallel_Poll())
728  ch |= PP_FLAG;
729 
730  if(uart_keyhit(0))
731  break;
732  if(GPIB_IO_RD(IFC) == 0)
733  {
734  ch |= IFC_FLAG;
735  break;
736  }
737 
738  switch(rx_state)
739  {
740  case GPIB_RX_START:
741  GPIB_PIN_FLOAT(NRFD);
742  GPIB_BUS_SETTLE(); // Let Data BUS settle
743  rx_state = GPIB_RX_WAIT_FOR_DAV_LOW;
744  break;
745 
746  case GPIB_RX_WAIT_FOR_DAV_LOW:
747  if ( GPIB_IO_RD(DAV) == 0 )
748  rx_state = GPIB_RX_DAV_IS_LOW;
749  break;
750 
751  case GPIB_RX_DAV_IS_LOW:
752  GPIB_IO_LOW(NRFD); // BUSY
753 
754  val = ~(GPIB_BUS_RD()); // Accept databyte
755  val &= 0xff;
756 
757  status = 0;
758  status |= (GPIB_IO_RD(EOI)) ? 0 : EOI_FLAG;
759  status |= (GPIB_IO_RD(IFC)) ? 0 : IFC_FLAG;
760  status |= (GPIB_IO_RD(ATN)) ? 0 : ATN_FLAG;
761  status |= (GPIB_IO_RD(SRQ)) ? 0 : SRQ_FLAG;
762  status |= (GPIB_IO_RD(REN)) ? 0 : REN_FLAG;
763 
764  if(status & ATN_FLAG)
765  val &= 0x7f;
766  else
767  val &= 0xff;
768 
769  ch |= (val | status);
770 
771  GPIB_PIN_FLOAT(NDAC); // Acknowledge Read
772  GPIB_BUS_SETTLE(); // Let Data BUS settle
773  gpib_timeout_set(HTIMEOUT);
774  rx_state = GPIB_RX_WAIT_FOR_DAV_HI;
775  break;
776 
777  case GPIB_RX_WAIT_FOR_DAV_HI:
778  if (GPIB_IO_RD(DAV) == 1)
779  rx_state = GPIB_RX_DAV_IS_HI;
780 
781  if (gpib_timeout_test())
782  {
783  ch |= TIMEOUT_FLAG;
784  rx_state = GPIB_RX_ERROR;
785  }
786  break;
787 
788  case GPIB_RX_DAV_IS_HI:
789  GPIB_IO_LOW(NDAC);
790  rx_state = GPIB_RX_FINISH; // DONE
791  break;
792 
793  case GPIB_RX_FINISH:
794  GPIB_IO_LOW(NDAC);
795  rx_state = GPIB_RX_DONE;
796  break;
797 
798  case GPIB_RX_ERROR:
799  GPIB_IO_LOW(NDAC);
800  GPIB_IO_LOW(NRFD);
801  rx_state = GPIB_RX_DONE;
802  break;
803 
804  case GPIB_RX_DONE:
805  break;
806  }
807  }
