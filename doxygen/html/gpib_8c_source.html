721 
722  GPIB_PIN_FLOAT(IFC);

727 
728  GPIB_BUS_IN();
729 
730  if(gpib_unread_f)
731  {
732  gpib_unread_f = 0;
733  return(gpib_unread_data);
734  }
735 
736  GPIB_PIN_FLOAT(NRFD);
737  GPIB_BUS_SETTLE(); // Let Data BUS settle
738  GPIB_IO_LOW(NDAC);
739 
740  rx_state = GPIB_RX_START;
741 
742  while(rx_state != GPIB_RX_DONE)
743  {
744  if(Parallel_Poll())
745  ch |= PP_FLAG;
746 
747  if(uart_keyhit(0))
748  break;
749  if(GPIB_IO_RD(IFC) == 0)
750  {
751  ch |= IFC_FLAG;
752  break;
753  }
754 
755  switch(rx_state)
756  {
757  case GPIB_RX_START:
758  GPIB_PIN_FLOAT(NRFD);
759  GPIB_BUS_SETTLE(); // Let Data BUS settle
760  rx_state = GPIB_RX_WAIT_FOR_DAV_LOW;
761  break;
762 
763  case GPIB_RX_WAIT_FOR_DAV_LOW:
764  if ( GPIB_IO_RD(DAV) == 0 )
765  rx_state = GPIB_RX_DAV_IS_LOW;
766  break;
767 
768  case GPIB_RX_DAV_IS_LOW:
769  GPIB_IO_LOW(NRFD); // BUSY
770 
771  val = ~(GPIB_BUS_RD()); // Accept databyte
772  val &= 0xff;
773 
774  status = 0;
775  status |= (GPIB_IO_RD(EOI)) ? 0 : EOI_FLAG;
776  status |= (GPIB_IO_RD(IFC)) ? 0 : IFC_FLAG;
777  status |= (GPIB_IO_RD(ATN)) ? 0 : ATN_FLAG;
778  status |= (GPIB_IO_RD(SRQ)) ? 0 : SRQ_FLAG;
779  status |= (GPIB_IO_RD(REN)) ? 0 : REN_FLAG;
780 
781  if(status & ATN_FLAG)
782  val &= 0x7f;
783  else
784  val &= 0xff;
785 
786  ch |= (val | status);
787 
789 
790 #if 0 // Disabled example
791  if((status & ATN_FLAG) && (ch & 0x7f) == SS80_MSA && talking == UNT)
792  {
793  DisablePPR();
794  }
795 #endif
796  GPIB_PIN_FLOAT(NDAC); // Acknowledge Read
797  GPIB_BUS_SETTLE(); // Let Data BUS settle
798  gpib_timeout_set(HTIMEOUT);
799  rx_state = GPIB_RX_WAIT_FOR_DAV_HI;
800  break;
801 
802  case GPIB_RX_WAIT_FOR_DAV_HI:
803  if (GPIB_IO_RD(DAV) == 1)
804  rx_state = GPIB_RX_DAV_IS_HI;
805 
806  if (gpib_timeout_test())
807  {
808  ch |= TIMEOUT_FLAG;
809  rx_state = GPIB_RX_ERROR;
810  }
811  break;
812 
813  case GPIB_RX_DAV_IS_HI:
814  GPIB_IO_LOW(NDAC);
815  rx_state = GPIB_RX_FINISH; // DONE
816  break;
817 
818  case GPIB_RX_FINISH:
819  GPIB_IO_LOW(NDAC);
820  rx_state = GPIB_RX_DONE;
821  break;
822 
823  case GPIB_RX_ERROR:
824  GPIB_IO_LOW(NDAC);
825  GPIB_IO_LOW(NRFD);
826  rx_state = GPIB_RX_DONE;
827  break;
828 
829  case GPIB_RX_DONE:
830  break;
831  }
832  }
