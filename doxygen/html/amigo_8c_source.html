213 
214

216  {
217  amigo_status[3] |= 0x40; // Write protect 0x40, reserved = 0x20
218  amigo_status[3] |= 0x20; // reserved = 0x20 ???
219  }
220 
221  if(Disk.dsj == 2)
222  {
223  amigo_status[0] = 0b00010011; // S1 error power on
224  amigo_status[3] |= 0x08; // F bit, power up
225  }
226  else if(Disk.Amigo_Errors || Disk.dsj == 1)
227  {
228  if(Disk.Amigo_Errors & ERR_GPIB)
229  amigo_status[0] = 0b00001010; // S1 error I/O error
