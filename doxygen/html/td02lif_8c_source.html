251 
252  // printf("td0_rle: len:%3d, type:%d\n", (int) len, (int)type);
253 
254  switch ( type )
255  {
256  case 0:
257 
258  if(len > max)
259  {
260  printf("td0_rle: type 0 len:%d > max:%d\n", len, max);
261  memcpy(dst,src,max);
262  result = -max;
263  error = 1;
264  }
265  else
266  {
267  memcpy(dst,src,len);
268  result = len;
269  }
270  break;
271 
272  case 1:
273  result = 0;
275  len = B2V_LSB(src,0,2) << 1;
276  if(len > max)
277  {
278  printf("td0_rle: type 1 len:%d > max:%d\n", len, max);
279  error = 1;
280  len = max;
281  }
282  while (len >= 2)
283  {
284  *dst++ = src[2];
285  *dst++ = src[3];
286  result += 2;
287  len -= 2;
288  }
289  break;
290  case 2:
291  result = 0;
292  do
293  {
294  if( !src[0] )
295  {
296  size = src[1];
297  src += 2;
298  len -= 2;
299 
300  if((result+size) > max)
301  {
302  printf("td0_rle: type 1 len:%d > max:%d\n",
303  result+size, max);
304  error = 1;
305  break;
306  }
307 
308  memcpy(dst,src,size);
309 
310  result += size;
311  dst += size;
312  src += size;
313  len -= size;
314  }
315  else
316  {
318  size = (1 << src[0]);
319 
321  repeat = src[1];
322 
323  src += 2;
324  len -= 2;
325 
326  while(repeat--)
327  {
328 
329  if((result+size) > max)
330  {
331  printf("td0_rle: type 1 len:%d > max:%d\n",
332  result+size, max);
333  error = 1;
334  break;
335  }
336 
337  memcpy(dst, src, size);
338 
339  result += size;
340  dst += size;
341  }
342 
343  src += size;
344  len -= size;
345  }
346 
347  }
348  while(len > 0);
349  if(len)
350  {
351  printf("td0_rle: type 1 len:%d < 0\n", len);
352  error = 1;
353  }
354  break;
355 
356  default:
357  printf("td0_rle error unsupported type:%02Xh\n", type);
358  result = -1;
359  break;
360  }
361  if(error)
362  return(-result);
363  return (result);
364 }
