117 void mmc_task (void)
118 {
119  BYTE s;
120 
121  if(mmc_us_timeout >= MMC_TIMER_TIC_US)
122  mmc_us_timeout -= MMC_TIMER_TIC_US;
123  else
124  mmc_us_timeout = 0;
125 
126  __mmc_task_timer -= MMC_TIMER_TIC_US;
127  if (__mmc_task_timer < 0)
128  {
129  __mmc_task_timer = 10000L; // 10,000us
130 
131  s = Stat;
132 
133  if (mmc_wp_status()) /* Write protected */
134  s |= STA_PROTECT;
135  else /* Write enabled */
136  s &= ~STA_PROTECT;
137 
138  if (mmc_ins_status()) /* Card inserted */
139  s &= ~STA_NODISK;
140  else /* Socket empty */
141  s |= (STA_NODISK | STA_NOINIT);
142  Stat = s; /* Update MMC status */
143  }
