88 uint8_t TWI_StartTransmission(const uint8_t SlaveAddress, const uint8_t TimeoutMS)
89 {
90  for (;;)
91  {
92  bool BusCaptured = false;
93  uint16_t TimeoutRemaining;
94 
95  TWCR = ((1 << TWINT) | (1 << TWSTA) | (1 << TWEN));
96 
97  TimeoutRemaining = (TimeoutMS * 100);
98  while (TimeoutRemaining-- && !(BusCaptured))
99  {
100  if (TWCR & (1 << TWINT))
101  {
102  switch (TWSR & TW_STATUS_MASK)
103  {
104  case TW_START:
105  case TW_REP_START:
106  BusCaptured = true;
107  break;
108  case TW_MT_ARB_LOST:
109  TWCR = ((1 << TWINT) | (1 << TWSTA) | (1 << TWEN));
110  continue;
111  default:
112  TWCR = (1 << TWEN);
113  return TWI_ERROR_BusFault;
114  }
115  }
116 
117  _delay_us(10);
118  }
119 
120  if (!(TimeoutRemaining))
121  {
122  TWCR = (1 << TWEN);
123  return TWI_ERROR_BusCaptureTimeout;
124  }
125 
126  TWDR = SlaveAddress;
127  TWCR = ((1 << TWINT) | (1 << TWEN));
128 
129  TimeoutRemaining = (TimeoutMS * 100);
130  while (TimeoutRemaining--)
131  {
132  if (TWCR & (1 << TWINT))
133  break;
134 
135  _delay_us(10);
136  }
137 
138  if (!(TimeoutRemaining))
139  return TWI_ERROR_SlaveResponseTimeout;
140 
141  switch (TWSR & TW_STATUS_MASK)
142  {
143  case TW_MT_SLA_ACK:
144  case TW_MR_SLA_ACK:
145  return TWI_ERROR_NoError;
146  default:
147  TWCR = ((1 << TWINT) | (1 << TWSTO) | (1 << TWEN));
148  return TWI_ERROR_SlaveNotReady;
149  }
150  }
151 }

158 bool TWI_SendByte(const uint8_t Byte)
159 {
160  TWDR = Byte;
161  TWCR = ((1 << TWINT) | (1 << TWEN));
162  while (!(TWCR & (1 << TWINT)));
163 
164  return ((TWSR & TW_STATUS_MASK) == TW_MT_DATA_ACK);

168 bool TWI_ReceiveByte(uint8_t* const Byte,
169 const bool LastByte)
170 {
171  uint8_t TWCRMask;
172 
173  if (LastByte)
174  TWCRMask = ((1 << TWINT) | (1 << TWEN));
175  else
176  TWCRMask = ((1 << TWINT) | (1 << TWEN) | (1 << TWEA));
177 
178  TWCR = TWCRMask;
179  while (!BIT_TST(TWCR, TWINT))
180  ;
181  *Byte = TWDR;
182 
183  uint8_t Status = (TWSR & TW_STATUS_MASK);
184 
185  return ((LastByte) ? (Status == TW_MR_DATA_NACK) : (Status == TW_MR_DATA_ACK));
186 }
