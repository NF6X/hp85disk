31 /*
32  * By default, the control register is set to:
33  * - TWEA: Automatically send acknowledge bit in receive mode.
34  * - TWEN: Enable the I2C system.
35  * - TWIE: Enable interrupt requests when TWINT is set.
36  */
37 #define TWCR_DEFAULT (_BV(TWEA) | _BV(TWEN) | _BV(TWIE))
38 
39 #define TWCR_NOT_ACK (_BV(TWINT) | _BV(TWEN) | _BV(TWIE))
40 #define TWCR_ACK (TWCR_NOT_ACK | _BV(TWEA))

42 void i2c_init(uint32_t speed) {
43  uint8_t sreg;
44  uint16_t temp;
45 
46  /* Store the status register and disable interrupts. */
47  sreg = SREG;
48  cli();
49 
50  /*
51  * From ATmega328P datasheet:
52  * SCL freq = (CPU Clock freq) / (16 + 2(TWBR) * (PrescalerValue))
53  *
54  * Which means:
55  * TWBR = ((CPU Clock freq) / (SCL freq) - 16) / (2 * (PrescalerValue))
56  *
57  * Disable the prescaler and set TWBR according to CPU freq and SCL freq.
58  */
59  TWSR &= ~(_BV(TWPS1) | _BV(TWPS0));
60  temp = ((F_CPU / speed) - 16) / (2 * 1);
61  if(temp & 0xff00)
62  printf("i2c_init prescale overflow\n");
63 
64  TWBR = temp;
65 
66  /*
67  * Active internal pull-up resistors for SCL and SDA.
68  * Their ports are SCL SDA
69  */
70 
71  GPIO_PIN_LATCH_HI(SCL); // Pull Up on
72  GPIO_PIN_LATCH_HI(SDA); // Pull Up on
73 
74  // PORTC |= _BV(PC5) | _BV(PC4);
75 
76  /* Enable interrupts via the control register. */
77  TWCR = TWCR_DEFAULT;
78 
79  /* Disable slave mode. */
80  TWAR = 0;
81 
82  /* Restore the status register. */
83  SREG = sreg;
84 }

86 void i2c_post(i2c_txn_t *t) {
87  uint8_t sreg;
88 
89  /* Reset transaction attributes. */
90  t->flags = 0;
91  t->opspos = 0;
92  t->next = NULL;
93 
94  sreg = SREG;
95  cli();
96 
97  /* Append transaction to linked list. */
98  if (txn == NULL) {
99  txn = t;
100  op = &txn->ops[0];
101 
102  /* Transmit START to kickstart things. */
103  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTA);
104  } else {
105  volatile i2c_txn_t *txn_ = txn;
106  while (txn_ != NULL) {
107  if (txn_->next != NULL) {
108  txn_ = txn_->next;
109  } else {
110  txn_->next = t;
111  break;
112  }
113  }
114  }
115 
116  SREG = sreg;
117 }

119 ISR(TWI_vect, ISR_BLOCK) {
120 
121  uint8_t status = TW_STATUS;
122 
123  /* This interrupt should only fire if there is something to do. */
124  assert(op != NULL);
125 
126  if ((op->address & _BV(0)) == TW_READ) {
127  /* Master Receiver mode. */
128  switch (status) {
129 
130  /* A START condition has been transmitted. */
131  case TW_START:
132  /* A repeated START condition has been transmitted. */
133  case TW_REP_START:
134  assert(op->buflen > 0);
135  op->bufpos = 0;
136  TWDR = op->address;
137  TWCR = TWCR_DEFAULT | _BV(TWINT);
138  break;
139 
140  /* Arbitration lost in SLA+R or NOT ACK bit. */
141  case TW_MR_ARB_LOST:
142  /* A START condition will be transmitted when the bus becomes free. */
143  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTA);
144  break;
145 
146  /* SLA+R has been transmitted; ACK has been received. */
147  case TW_MR_SLA_ACK:
148  if (op->buflen == 1) {
149  TWCR = TWCR_NOT_ACK;
150  } else {
151  TWCR = TWCR_ACK;
152  }
153  break;
154 
155  /* SLA+R has been transmitted; NOT ACK has been received. */
156  case TW_MR_SLA_NACK:
157  txn->flags = I2C_TXN_DONE | I2C_TXN_ERR;
158  goto next_txn;
159 
160  /* Data byte has been received; ACK has been returned. */
161  case TW_MR_DATA_ACK:
162  op->buf[op->bufpos++] = TWDR;
163  if (op->bufpos+1 == op->buflen) {
164  TWCR = TWCR_NOT_ACK;
165  } else {
166  TWCR = TWCR_ACK;
167  }
168  break;
169 
170  /* Data byte has been received; NOT ACK has been returned. */
171  case TW_MR_DATA_NACK:
172  op->buf[op->bufpos++] = TWDR;
173  goto next_op;
174 
175  default:
176  assert(0 && "unknown status in master receiver mode");
177  }
178  } else {
179  /* Master Transmitter mode. */
180  switch (status) {
181 
182  /* A START condition has been transmitted. */
183  case TW_START:
184  /* A repeated START condition has been transmitted. */
185  case TW_REP_START:
186  assert(op->buflen > 0);
187  op->bufpos = 0;
188  TWDR = op->address;
189  TWCR = TWCR_DEFAULT | _BV(TWINT);
190  break;
191 
192  /* Arbitration lost in SLA+W or data bytes. */
193  case TW_MT_ARB_LOST:
194  /* A START condition will be transmitted when the bus becomes free. */
195  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTA);
196  break;
197 
198  /* SLA+W has been transmitted; ACK has been received. */
199  case TW_MT_SLA_ACK:
200  TWDR = op->buf[op->bufpos++];
201  TWCR = TWCR_DEFAULT | _BV(TWINT);
202  break;
203 
204  /* SLA+W has been transmitted; NOT ACK has been received. */
205  case TW_MT_SLA_NACK:
206  txn->flags = I2C_TXN_DONE | I2C_TXN_ERR;
207  goto next_txn;
208 
209  /* Data byte has been transmitted; ACK has been received. */
210  case TW_MT_DATA_ACK:
211  if (op->bufpos < op->buflen) {
212  TWDR = op->buf[op->bufpos++];
213  TWCR = TWCR_DEFAULT | _BV(TWINT);
214  break;
215  }
216 
217  /* No more bytes left to transmit... */
218  goto next_op;
219 
220  /* Data byte has been transmitted; NOT ACK has been received. */
221  case TW_MT_DATA_NACK:
222  if (op->bufpos < op->buflen) {
223  /* There were more bytes left to transmit! */
224  txn->flags = I2C_TXN_DONE | I2C_TXN_ERR;
225  goto next_txn;
226  }
227 
228  goto next_op;
229 
230  default:
231  assert(0 && "unknown status in master transmitter mode");
232  }
233  }
234 
235  return;
236 
237 next_op:
238  /*
239  * Advance to next operation in transaction, if possible.
240  */
241  if (++(txn->opspos) < txn->opslen) {
242  op = &txn->ops[txn->opspos];
243 
244  /* Repeated start. */
245  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTA);
246  return;
247  }
248 
249  /* No more operations, mark transaction as done. */
250  txn->flags = I2C_TXN_DONE;
251 
252 next_txn:
253  /*
254  * Advance to next transaction, if possible.
255  */
256  if (txn->next != NULL) {
257  txn = txn->next;
258  op = &txn->ops[0];
259 
260  /* Repeated start. */
261  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTA);
262  return;
263  }
264 
265  txn = NULL;
266  op = NULL;
267 
268  /* No more transaction, transmit STOP. */
269  TWCR = TWCR_DEFAULT | _BV(TWINT) | _BV(TWSTO);
270 }
