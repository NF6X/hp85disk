100 uint16_t uart_ubr(uint32_t baud, int *u2x, uint32_t *actual)
101 {
102  double div;
103  uint32_t ubr_regi;
104 
105  // Calculating Baud Rate
106  // (U2X = 0)
107  // BAUD = Fosc/(16*(UBRn+1))
108  // UBRn = Fosc/(16*Baud) -1
109  // (U2X = 1)
110  // BAUD = Fosc/(8*(UBRn+1))
111  // UBRn = Fosc/(8*Baud) -1
112 
114  *u2x = 1;
115  div = 8;
116  ubr_regi = round( ((double)F_CPU/(div*(double)baud)) - 1.0 );
117 
118  // For lower baud rates use 16 divider if the UBRR register overflows
119  // URBRR register is only a 12 bit register!
120  if(ubr_regi > 4095)
121  {
123  *u2x = 0;
124  div = 16.0;
125  ubr_regi = round( ((double)F_CPU/(div*(double)baud)) - 1.0 );
126  }
127  //overflow, baud rate was too low - so we clip to maximum allowed
128  if(ubr_regi > 4095)
129  ubr_regi = 4095;
130 
131  *actual = ((double)F_CPU/(div*((double)(ubr_regi+1))));
132 
133  return(ubr_regi);
134 }
