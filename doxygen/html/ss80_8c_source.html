495 int SS80_locate_and_write(void)
496 {
497  DWORD total_bytes; // Keeps track of data actually written to disk
498  int chunk, count, len;
499  int io_skip;
500  uint16_t status;
501  uint32_t Address = SS80_Blocks_to_Bytes(SS80s->AddressBlocks);
502 
503  io_skip = 0;
504 
505 #if SDEBUG
506  if(debuglevel & 32)
507  printf("[SS80 Locate and Write at %08lXH(%lXH)]\n", (long)Address, (long)SS80s->Length);
508 #endif
509 
510  SS80s->qstat = 0;
511 
512  if( GPIB_IO_RD(IFC) == 0)
513  return(IFC_FLAG);
514 
515  if( SS80_cmd_seek() )
516  {
517  SS80s->Errors |= ERR_WRITE;
518  io_skip = 1;
519  }
520 
521  count = SS80s->Length;
522  total_bytes = 0;
523 
524  status = 0;
525 
526  while(count > 0) // Loop until we have received Length sectors or EOI
527  {
528  if( GPIB_IO_RD(IFC) == 0)
529  {
530  return(IFC_FLAG);
531  }
532 
533  if(count > 256)
534  chunk = 256;
535  else
536  chunk = count;
537 
538  Mem_Clear(gpib_iobuff);
539 
540 #if SDEBUG
541  if(debuglevel & 128)
542  gpib_timer_elapsed_begin();
543 #endif
544  len = gpib_read_str(gpib_iobuff, (UINT) chunk, &status);
545 
546 #if SDEBUG
547  if(debuglevel & 128)
548  gpib_timer_elapsed_end("GPIB Read");
549 #endif
550 
551  if( len != chunk)
552  {
553  if(status & ERROR_MASK)
554  {
555  if(debuglevel & 1)
556  printf("[GPIB Read Error]\n");
557  SS80s->Errors |= ERR_WRITE;
558  SS80s->qstat = 1;
559  break;
560  }
561  if(!len && (EOI_FLAG & status))
562  break;
563  }
564 
565 
566  if(!io_skip)
567  {
568  if(len)
569  {
570  int len2;
571 #if SDEBUG
572  if(debuglevel & 64)
573  gpib_timer_elapsed_begin();

583  if(mmc_wp_status())
584  SS80s->Errors |= ERR_WP;
585  SS80s->qstat = 1;

595 #endif

597  }
598 
599  if(len2 >= len)
600  total_bytes += len2;
601  else
602  total_bytes += len;
603  } // len
604  } // io_skip
605 
606  count -= len;
607 
608  if(status & EOI_FLAG)
609  break;
610  }
611 
612  if(count > 0)
613  {
614  if(debuglevel & 1)
615  printf("[SS80 Locate and Write DID NOT FINISH]\n");
616  }
617  else
618  {
619 #if SDEBUG
620  if(debuglevel & 32)
621  printf("[SS80 Locate and Write Wrote Total(%lxH)]\n", total_bytes);
622 #endif
623  }

625  return ( status & ERROR_MASK );
